```cpp
void drawpixel(int x, int y, int color) {}
void BresenhamLine(int x0, int y0, int x1, int y1, int color) {
    int x, y, dx, dy;
    float k;
    int e;
    dx = x1 - x0;
    dy = y1 - y0;
    k = dy / dx;
    e = -dx;
    x = x0;
    y = y0;
    for (int i = 0; i <= dx; i++) {
        drawpixel(x, y, color);
        x++;
        e = e + 2 * dy;
        if (e >= 0) {
            y++;
            e = e - 2 * dx;
        }
    }
}
```
